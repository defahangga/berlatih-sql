MEMBUAT TABEL

    a. Tabel users
       syntax :
       CREATE TABLE users (
       -> id int primary key auto_increment,
       -> name varchar(255),
       -> email varchar(255),
       -> password varchar(255)
       -> );

    b. Tabel items
       syntax :
       CREATE TABLE items (
       -> id int primary key auto_increment,
       -> name varchar(255),
       -> description varchar(255),
       -> price int,
       -> stock int,
       -> category_id int
       -> );

    c. Tabel categories
       syntax :
       CREATE TABLE categories (
       -> id int primary key auto_increment,
       -> name varchar(255)
       -> );

       MENAMBAHKAN FOREIGN KEY DI TABEL ITEMS
